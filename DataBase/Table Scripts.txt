CREATE TABLE SupplierProduct
( 
   Id NUMBER(20) NOT NULL PRIMARY KEY,
   SupplierId NUMBER(20),
   ProductId NUMBER(20),
   Price DECIMAL(19,2),
   Currency VARCHAR(3),
   SKU VARCHAR(50),
   Priority INT,
   CreatedTime TIMESTAMP WITH LOCAL TIME ZONE,
   CreatedBy VARCHAR(20),
   UPDATEDTIME TIMESTAMP WITH LOCAL TIME ZONE,
   UPDATEDBY VARCHAR(20),
   IsActive CHAR,
   OfferBucketSize INT
);

CREATE TABLE ProductDomiciles
(
    Id NUMBER(20) NOT NULL PRIMARY KEY,
    ProductId NUMBER(20),
    Country VARCHAR(3),
    SKU VARCHAR(50),
    Currency VARCHAR(3),
    MSRP DECIMAL(19,2)
)

CREATE TABLE Products
(
    Id NUMBER(20) NOT NULL PRIMARY KEY,
    Manufacturer VARCHAR(100),
    Name VARCHAR(200),
    DESCRIPTION VARCHAR(250),
    EAN VARCHAR(50),
    GTIN VARCHAR(50),
    UOM VARCHAR(50),
    CreatedTime TIMESTAMP WITH LOCAL TIME ZONE,
    CreatedBy VARCHAR(20),
    UPDATEDTIME TIMESTAMP WITH LOCAL TIME ZONE,
    UPDATEDBY VARCHAR(20),
    IsActive CHAR
);

CREATE TABLE Suppliers
(
    Id NUMBER(20) NOT NULL PRIMARY KEY,
    DOMAIN VARCHAR(20),
    SUPPLIERTYPE VARCHAR(50),
    Name VARCHAR(200),
    Priority INT,
    Currency VARCHAR(3),
    CreatedTime TIMESTAMP WITH LOCAL TIME ZONE,
    CreatedBy VARCHAR(20),
    UPDATEDTIME TIMESTAMP WITH LOCAL TIME ZONE,
    UPDATEDBY VARCHAR(20),
    IsActive CHAR,
    PARENTSUPPLIERID NUMBER(20)
);

ALTER TABLE SupplierProduct
ADD FOREIGN KEY (SupplierId) REFERENCES Suppliers(Id);

ALTER TABLE SupplierProduct
ADD FOREIGN KEY (ProductId) REFERENCES Products(Id);

ALTER TABLE ProductDomiciles
ADD FOREIGN KEY (ProductId) REFERENCES Products(Id);
